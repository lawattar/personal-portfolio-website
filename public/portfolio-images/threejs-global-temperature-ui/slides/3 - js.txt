import * as THREE from "https://esm.sh/three";
import { OrbitControls } from "https://cdn.jsdelivr.net/npm/three@0.161/examples/jsm/controls/OrbitControls.js";

function getStarfield({ numStars = 500, sprite } = {}) {
    function randomSpherePoint() {
        const radius = Math.random() * 25 + 25;
        const u = Math.random();
        const v = Math.random();
        const theta = 2 * Math.PI * u;
        const phi = Math.acos(2 * v - 1);
        let x = radius * Math.sin(phi) * Math.cos(theta);
        let y = radius * Math.sin(phi) * Math.sin(theta);
        let z = radius * Math.cos(phi);

        return {
            pos: new THREE.Vector3(x, y, z),
            hue: 0.6,
            minDist: radius,
        };
    }

    const verts = [];
    const colors = [];
    const positions = [];
    let col;
    for (let i = 0; i < numStars; i += 1) {
        let p = randomSpherePoint();
        const { pos, hue } = p;
        positions.push(p);
        col = new THREE.Color().setHSL(hue, 0.2, Math.random());
        verts.push(pos.x, pos.y, pos.z);
        colors.push(col.r, col.g, col.b);
    }

    const geo = new THREE.BufferGeometry();
    geo.setAttribute("position", new THREE.Float32BufferAttribute(verts, 3));
    geo.setAttribute("color", new THREE.Float32BufferAttribute(colors, 3));

    const mat = new THREE.PointsMaterial({
        size: 0.2,
        vertexColors: true,
        map: sprite,
    });

    const points = new THREE.Points(geo, mat);
    return points;
}

const scene = new THREE.Scene();
const camera = new THREE.PerspectiveCamera(45, window.innerWidth / window.innerHeight, 0.1, 1000);
camera.position.set(0, 0, 3.5);

const renderer = new THREE.WebGLRenderer({ antialias: true });
renderer.setSize(window.innerWidth, window.innerHeight);
renderer.setPixelRatio(window.devicePixelRatio);
document.body.appendChild(renderer.domElement);

const orbitCtrl = new OrbitControls(camera, renderer.domElement);
orbitCtrl.enableDamping = true;
orbitCtrl.dampingFactor = 0.05;

const textureLoader = new THREE.TextureLoader();

const textureUrls = {
    "circle.png": "https://raw.githubusercontent.com/bobbyroe/vertex-earth/main/src/circle.png",
    "00_earthmap1k.jpg": "https://raw.githubusercontent.com/bobbyroe/vertex-earth/main/src/00_earthmap1k.jpg",
    "01_earthbump1k.jpg": "https://raw.githubusercontent.com/bobbyroe/vertex-earth/main/src/01_earthbump1k.jpg",
    "02_earthspec1k.jpg": "https://raw.githubusercontent.com/bobbyroe/vertex-earth/main/src/02_earthspec1k.jpg",
    "03_earthlights1k.jpg": "https://raw.githubusercontent.com/bobbyroe/vertex-earth/main/src/03_earthlights1k.jpg",
    "04_rainbow1k.jpg": "https://raw.githubusercontent.com/bobbyroe/vertex-earth/main/src/04_rainbow1k.jpg"
};

const loadedTextures = {};
for (const key in textureUrls) {
    loadedTextures[key] = textureLoader.load(textureUrls[key]);
}

const starSprite = loadedTextures["circle.png"];
const colorMap = loadedTextures["00_earthmap1k.jpg"];
const elevMap = loadedTextures["01_earthbump1k.jpg"];
const alphaMap = loadedTextures["02_earthspec1k.jpg"];

const globeGroup = new THREE.Group();
scene.add(globeGroup);

const geo = new THREE.IcosahedronGeometry(1, 10);
const mat = new THREE.MeshBasicMaterial({
    color: 0x002d39,
    wireframe: true,
});
const cube = new THREE.Mesh(geo, mat);
globeGroup.add(cube);

const detail = 120;
const pointsGeo = new THREE.IcosahedronGeometry(1, detail);

const vertexShader = `
    uniform float size;
    uniform sampler2D elevTexture;

    varying vec2 vUv;
    varying float vVisible;

    void main() {
        vUv = uv;
        vec4 mvPosition = modelViewMatrix * vec4( position, 1.0 );
        float elv = texture2D(elevTexture, vUv).r;
        vec3 vNormal = normalMatrix * normal;
        vVisible = step(0.0, dot( -normalize(mvPosition.xyz), normalize(vNormal)));
        mvPosition.z += 0.35 * elv;
        gl_PointSize = size;
        gl_Position = projectionMatrix * mvPosition;
    }
`;
const fragmentShader = `
    uniform sampler2D colorTexture;
    uniform sampler2D alphaTexture;

    varying vec2 vUv;
    varying float vVisible;

    void main() {
        if (floor(vVisible + 0.1) == 0.0) discard;
        float alpha = 1.0 - texture2D(alphaTexture, vUv).r;
        vec3 color = texture2D(colorTexture, vUv).rgb;
        gl_FragColor = vec4(color, alpha);
    }
`;
const uniforms = {
    size: { type: "f", value: 4.0 },
    colorTexture: { type: "t", value: colorMap },
    elevTexture: { type: "t", value: elevMap },
    alphaTexture: { type: "t", value: alphaMap }
};
const pointsMat = new THREE.ShaderMaterial({
    uniforms: uniforms,
    vertexShader,
    fragmentShader,
    transparent: true
});

const points = new THREE.Points(pointsGeo, pointsMat);
globeGroup.add(points);

const hemiLight = new THREE.HemisphereLight(0xffffff, 0x080820, 3);
scene.add(hemiLight);

const stars = getStarfield({ numStars:4500, sprite: starSprite });
scene.add(stars);

let isSpinning = true; // Control variable for globe spinning

function animate() {
    renderer.render(scene, camera);
    if (isSpinning) {
      //i am speed
        globeGroup.rotation.y += 0.002;
    }
    requestAnimationFrame(animate);
    orbitCtrl.update();
};
animate();

window.addEventListener('resize', function () {
    camera.aspect = window.innerWidth / window.innerHeight;
    camera.updateProjectionMatrix();
    renderer.setSize(window.innerWidth, window.innerHeight);
}, false);

function updateSelectedButton(mapType, selectedTextureName) {
    const buttonGroup = document.querySelector(`#texture-menu .menu-section .button-group button[data-map-type="${mapType}"]`).closest('.button-group');
    if (buttonGroup) {
        buttonGroup.querySelectorAll('button').forEach(button => {
            button.classList.remove('selected');
        });
        const selectedButton = buttonGroup.querySelector(`button[data-texture-name="${selectedTextureName}"]`);
        if (selectedButton) {
            selectedButton.classList.add('selected');
        }
    }
}

updateSelectedButton('color', '00_earthmap1k.jpg');
updateSelectedButton('elev', '01_earthbump1k.jpg');

document.getElementById('texture-menu').addEventListener('click', (event) => {
    const target = event.target;
    if (target.tagName === 'BUTTON') {
        const mapType = target.dataset.mapType;
        const textureName = target.dataset.textureName;

        if (loadedTextures[textureName]) {
            switch (mapType) {
                case 'color':
                    pointsMat.uniforms.colorTexture.value = loadedTextures[textureName];
                    break;
                case 'elev':
                    pointsMat.uniforms.elevTexture.value = loadedTextures[textureName];
                    break;
            }
            pointsMat.needsUpdate = true;
            updateSelectedButton(mapType, textureName);
        }
    }
});

globeGroup.add(markerGroup);

const raycaster = new THREE.Raycaster();
const mouse = new THREE.Vector2();
let currentMarker = null; // The marker for the city
let INTERSECTED = null; 

const infoOverlay = document.getElementById('info-overlay');
const overlayLocation = document.getElementById('overlay-location');
const overlayTemp = document.getElementById('overlay-temp');
const closeOverlayButton = document.getElementById('close-overlay');
const cityNameInput = document.getElementById('city-name-input');

//longitude and latitude to Cartesian coordinates
function lonLatToCartesian(lon, lat, radius) {
    // Convert latitude and longitude to radians
    const latRad = lat * Math.PI / 180;
    //longitude by 180 degrees
    const lonAdjusted = lon; 
    const lonRad = lonAdjusted * Math.PI / 180;

    // X-axis  right
    // Y-axis  up (North pole)
    // Z-axis  forward
    //no negative = flipped
    const x = radius * Math.cos(latRad) * -Math.cos(lonRad);
    const y = radius * Math.sin(latRad);
    const z = radius * Math.cos(latRad) * Math.sin(lonRad);

    return new THREE.Vector3(x, y, z);
}
//dont want error
async function loadCityTemperature() {
    const cityName = cityNameInput.value.trim();
    if (!cityName) {
        alert('Please enter a city name.');
        return;
    }

    // Clear previous marker and hide overlay if a new search is performed
    if (currentMarker) {
        if (INTERSECTED === currentMarker) { 
            INTERSECTED.material.color.setHex(0xffff00); 
            INTERSECTED = null;
        }
        markerGroup.remove(currentMarker);
        currentMarker = null;
    }
    infoOverlay.style.display = 'none';
//more error and city name to coords
    try {
        const geoResponse = await fetch(`https://geocoding-api.open-meteo.com/v1/search?name=${cityName}&count=1&language=en&format=json`);
        if (!geoResponse.ok) {
            throw new Error(`Failed to get coordinates for city: ${geoResponse.status} ${geoResponse.statusText}`);
        }
        const geoData = await geoResponse.json();

        if (!geoData.results || geoData.results.length === 0) {
            alert('Could not find coordinates for given city. Please check city name.');
            return;
        }

        let { latitude: lat, longitude: lon, name } = geoData.results[0];

        // The longitude adjustment is with lonLatToCartesian, so no negative here?
        //coords to weather
        const weatherResponse = await fetch(`https://api.open-meteo.com/v1/forecast?latitude=${lat}&longitude=${lon}&current=temperature_2m&temperature_unit=fahrenheit&forecast_days=1`);
        if (!weatherResponse.ok) {
            throw new Error(`Failed to fetch temperature data: ${weatherResponse.status} ${weatherResponse.statusText}`);
        }
        const weatherData = await weatherResponse.json();
        const temperature = weatherData.current.temperature_2m;

        const markerGeometry = new THREE.SphereGeometry(0.03, 16, 16);
        const markerMaterial = new THREE.MeshBasicMaterial({ color: 0xffff00 });
        const position = lonLatToCartesian(lon, lat, 1.1); // Radius 1 (globe) + 0.05 offset since clipping in elevation map

        const marker = new THREE.Mesh(markerGeometry, markerMaterial);
        marker.position.copy(position);
        marker.userData = {
            type: 'cityTempMarker',
            locationName: name,
            temperature: temperature,
        };
        markerGroup.add(marker);
        currentMarker = marker; // Set the current city marker

        isSpinning = false; // Stop the globe spinning since annoying and menu doesnt rotate cuz idk how

        // Orient camera normal to the new markers location
        const markerWorldPosition = new THREE.Vector3();
        marker.getWorldPosition(markerWorldPosition); 

        orbitCtrl.target.copy(markerWorldPosition);

        const directionFromOriginToMarker = markerWorldPosition.clone().normalize();
        const cameraDistance = 3.5; 
        camera.position.copy(directionFromOriginToMarker.multiplyScalar(cameraDistance));

        orbitCtrl.update(); 

        // Display info overlay
        overlayLocation.textContent = marker.userData.locationName;
        overlayTemp.textContent = `Temperature: ${marker.userData.temperature}°F`;
        requestAnimationFrame(() => {
        const markerScreenPosition = new THREE.Vector3();
        markerScreenPosition.copy(marker.position);
        globeGroup.localToWorld(markerScreenPosition);
        markerScreenPosition.project(camera);

        const x = (markerScreenPosition.x * 0.5 + 0.5) * window.innerWidth;
        const y = (-markerScreenPosition.y * 0.5 + 0.5) * window.innerHeight;

        infoOverlay.style.left = `${x}px`;
        infoOverlay.style.top = `${y}px`;
        infoOverlay.style.display = 'block';
        });
    } catch (error) {
        console.error('Error loading city temperature data:', error);
        alert(`Error: ${error.message}. Please ensure you enter a valid city name.`);
    }
}

showTempButton.addEventListener('click', loadCityTemperature);

function onMouseMove(event) {
    mouse.x = (event.clientX / window.innerWidth) * 2 - 1;
    mouse.y = -(event.clientY / window.innerHeight) * 2 + 1;

    raycaster.setFromCamera(mouse, camera);

    const inverseGlobeMatrix = new THREE.Matrix4().copy(globeGroup.matrixWorld).invert();
    const transformedRayOrigin = raycaster.ray.origin.clone().applyMatrix4(inverseGlobeMatrix);
    const transformedRayDirection = raycaster.ray.direction.clone().transformDirection(inverseGlobeMatrix);
    const transformedRaycaster = new THREE.Raycaster(transformedRayOrigin, transformedRayDirection);

    const intersects = transformedRaycaster.intersectObjects(markerGroup.children);

    if (intersects.length > 0 && intersects[0].object.userData.type === 'cityTempMarker') {
        const hoveredObject = intersects[0].object;
        if (INTERSECTED !== hoveredObject) {
            if (INTERSECTED) {
                INTERSECTED.material.color.setHex(0xffff00);
            }
            INTERSECTED = hoveredObject;
            INTERSECTED.material.color.setHex(0x00ff00);
        }
    } else {
        if (INTERSECTED) {
            INTERSECTED.material.color.setHex(0xffff00);
            INTERSECTED = null;
        }
    }
}

renderer.domElement.addEventListener('mousemove', onMouseMove, false);

closeOverlayButton.addEventListener('click', () => {
    infoOverlay.style.display = 'none';
    if (currentMarker) {
        currentMarker.material.color.setHex(0xffff00);
        if (INTERSECTED === currentMarker) {
            INTERSECTED = null;
        }
    }
});
